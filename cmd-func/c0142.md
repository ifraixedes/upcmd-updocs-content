##### Showcase a extended color print cmd, it is a improved cmd to support color printing for both plain text message, template rendering message and object


### Demo








[source](https://github.com/upcmd/up/blob/master/tests/functests/c0142.yml)

##### Main task yaml file
```
    tasks:
    - name: task
      task:
      - func: cmd
        vars:
          person:
            name: tom
            age: 18
          objectname: person
        do:
        - name: colorPrint
          cmd:
            msg: hello, world
            bg: blue
            fg: white
        - name: colorPrint
          cmd:
            msg: hello, world
            bg: yellow
        - name: colorPrint
          cmd:
            msg: hello, world
            fg: white
        - name: colorPrint
          cmd:
            msg: hello, world
        - name: colorPrint
          cmd:
            msg: '{{.person.name}}: {{.person.age}}'
            fg: blue
            bg: red
        - name: colorPrint
          cmd:
            object: 'person'
            fg: blue
            bg: black
        - name: colorPrint
          cmd:
            object: '{{.objectname}}'
            fg: blue
            bg: green
        - name: colorPrint
          cmd:
            object: 'person'
            msg: 'person'
            fg: blue
            bg: black
    
```
