##### Test the block-func being used in callee


### Demo








[source](https://github.com/upcmd/up/blob/master/tests/functests/c0160.yml)

##### Main task yaml file
```
    tasks:
    - name: task
      task:
      - func: call
        desc: |
          test var overriding in sub_task
        dvars:
        - name: aaa
          value: var_a_from_task
        do:
        - sub_task
    - name: sub_task
      task:
      - func: cmd
        desc: |
          inspect if the correct parameter has been passed in correctly
          without default value
        do:
        - name: assert
          cmd:
          - '{{eq .aaa "var_a_from_task"}}'
      - func: block
        do:
        - func: cmd
          desc: |
            inspect if the correct parameter has been passed in correctly
            without default value
            in block func
          do:
          - name: assert
            cmd:
            - '{{eq .aaa "var_a_from_task"}}'
      - func: cmd
        desc: |
          inspect if the correct parameter has been passed in correctly
          with default value
        vars:
          aaa: var_a_from_sub_task
        do:
        - name: assert
          cmd:
          - '{{eq .aaa "var_a_from_task"}}'
      - func: block
        vars:
          aaa: var_a_from_sub_task
        do:
        - func: cmd
          desc: |
            inspect if the correct parameter has been passed in correctly
            with default value
            in block func
          do:
          - name: assert
            cmd:
            - '{{eq .aaa "var_a_from_task"}}'
    
```
